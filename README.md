# django-url-shorter
simple django url Shorter web app

how to use:

    python -m venv venv
    or 
    python -m venv venv


activate:

      ./venv/Scripts/activate
      or 
      source .\venv\Scripts\activate

run:

      python manage.py migrate
      python manage.py runserver
